asset my_fish;
asset circle;
asset my_num;
asset my_name;
set my_num.value = 23;
set my_name.text = "Hello";

var number my_fib;

function number fibonacci (number n) {
	var number first;
	var number second;
	var number iterator;
	var number result;

	set result = 1;
	set first = 1;
	set second = 1;
	set iterator = 3;

	if (n == 1) {
		return 0;
	}

	if (n == 2 or n == 3) {
		return 1;
	}

	while (iterator <= n) {
		set result = first + second;
		set first = second;
		set second = result;
		set iterator = iterator + 1;
	}

	return result;
}

empty instructions {
	set my_fib = call fibonacci(5);
	set my_num.value = my_fib;
	set my_fish.color = green;
	
	if (my_fish.size >= 100) {
		do circle.spin;
		set my_name.value = "world";
	}

	for (var number loops; set loops = 0; loops < call fibonacci(6) + my_num.value - my_fib; set loops = loops + 1) {
		do my_fish.set_position(2+3, my_fib - 2);
		do my_num.move_x(10+5);
	}
}
